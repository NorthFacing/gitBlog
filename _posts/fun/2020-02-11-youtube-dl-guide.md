```shell
# 查看可以下载的格式列表
youtube-dl -F url
# 下载指定的格式
youtube-dl 
# 下载质量最好的m4a格式音频
youtube-dl -f 'bestaudio[ext=m4a]' url
# 下载质量最好的mp4视频以及质量最好的m4a格式音频，并合并
youtube-dl -f 'bestvideo[ext=mp4]+bestaudio[ext=m4a]/mp4' url
```
